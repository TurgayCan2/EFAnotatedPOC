#CurrentStateCallInfo ddl script

CREATE TABLE [dbo].[CURRENT_STATE_CALL_INFO]
(
[ID] BIGINT NOT NULL PRIMARY KEY IDENTITY,
[CALL_ID] VARCHAR(50) NULL,
[AGENT_ID] VARCHAR(50) NULL,
[TN_CUSTOMER_ID] VARCHAR(50) NOT NULL
)

#Course ddl script

CREATE TABLE [dbo].[T_COURSE] (
    [ID]   INT          IDENTITY (1, 1) NOT NULL,
    [NAME] VARCHAR (50) NOT NULL,
    PRIMARY KEY CLUSTERED ([ID] ASC)
);

#Student ddl script

CREATE TABLE [dbo].[T_STUDENT] (
    [ID]        BIGINT       IDENTITY (1, 1) NOT NULL,
    [NAME]      VARCHAR (50) NULL,
    [COURSE_ID] INT          NULL,
    PRIMARY KEY CLUSTERED ([ID] ASC),
    CONSTRAINT [FK_T_STUDENT_T_COURSE] FOREIGN KEY ([COURSE_ID]) REFERENCES [dbo].[T_COURSE] ([ID])
);
